import java.util.Scanner;
public class j {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		String c[] = {"", "紅色", "綠色", "藍色", "橘色", "黃色"};
		int    com = 0, man = 0 , ya = 0;
		for(int  i = 0; i < 5;){
			System.out.printf("第 %d 次：\n請選擇顏色 (1：紅色，2：綠色，3：藍色，4：橘色，5：黃色)：", ++i);
			System.out.printf("電腦選擇顏色：%s\t玩家選擇顏色：%s\t玩家答%s\n",
			 c[com = (int)(Math.random() * 5) + 1],  c[man = sc.nextInt()] ,(com == man ? "對" : "錯"));
			if(com == man) ya++;
		}
		System.out.printf("\n遊戲結束：玩家總共答對 %d 次。", ya);
	}
}
